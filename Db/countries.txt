#Countries

#Countries must be uniquely named. Tag must be a combination of three caracters from A to Z.

#picture = "<path>"       Picture displayed on the Country Selection Screen (default is "gfx\scenarios\picture.bmp" if not specified).
#color = <name>           Color used for the display on the political map.
#techgroup                For techspeed. Types are: latin, orthodox, muslim, china or exotic (default is exotic if not specified).
#leader_language = <tag>  Entry in the leader names DB file for random name of the leaders.
#new_colony = <tag>       Entry in new_colony section of the specific countries DB file for the beginning of the name of the new cities if the province is not specified in the colony names DB file (colony name is the default city name in provinces DB file if not specified).
#gfx = { }                Tag for army and navy sprites and random names in corresponding DB file for the display on the map.
#policy = { }             Default policy settings for revolter and if not specified in the country definition file (default values are average if not specified).
#elector = X             Number of votes for the HRE Emperor election (default is 0 and not involved if not specified).
#history = { }            Future use...
