# Numbering scheme: First a fixed number 717 specifying a range. Then a four
# digit number for the province. Finally a two digit number for the different
# events for that province.

# Main action events to change provinces to reformed during Jean Calvin.
# These won't happen by themselves, they are intended to (perhaps) be
# triggered by the helper events.

event = {
    id = 717033710
    random = no
    province = 337
    name = "Friesen converted to reformed"
    desc = "As a result of the spread of the reformation, the province is now considered as being of the reformed faith."
    action_a = {
        name = "OK"
        command = { type = provincereligion which = 337 value = reformed }
    }
}
